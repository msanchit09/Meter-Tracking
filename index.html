        const form = document.getElementById('dataForm');

        // IMPORTANT: Replace this with your deployed Google Apps Script Web App URL

        const WEB_APP_URL = 'https://script.google.com/macros/s/AKfycbwU6kErgNnhfAQHkywgficAvmH-EDREQl9m05uJKkDq6YXqe8AYMjYBI3aAOHc8PMXu/exec'; // Example: https://script.google.com/macros/s/AKfyc.../exec


        form.addEventListener('submit', async (event) => {

            event.preventDefault(); // Prevent default form submission


            messageBox.style.display = 'none'; // Hide previous messages

            messageBox.className = 'message-box'; // Reset message box classes


            // Show loading state

            submitBtn.disabled = true;

            submitBtn.innerHTML = '<span class="loading-spinner"></span> Submitting...';


            const formData = new FormData(form);

            const data = {};

            for (let [key, value] of formData.entries()) {

                data[key] = value;

            }


            try {

                const response = await fetch(WEB_APP_URL, {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/x-www-form-urlencoded', // Or 'application/json' if you change Apps Script to parse JSON

                    },

                    body: new URLSearchParams(data).toString(), // Form data sent as URL-encoded string

                    mode: 'cors' // Ensure CORS is enabled

                });


                if (!response.ok) {

                    throw new Error(`HTTP error! status: ${response.status}`);

                }


                const result = await response.json(); // Assuming Apps Script returns JSON


                if (result.status === 'success') {

                    messageBox.classList.add('success');

                    messageBox.textContent = 'Data submitted successfully!';

                    form.reset(); // Clear the form

                } else {

                    throw new Error(result.message || 'Failed to submit data.');

                }

            } catch (error) {

                messageBox.classList.add('error');

                messageBox.textContent = `Error: ${error.message}`;

                console.error('Submission error:', error);

            } finally {

                submitBtn.innerHTML = 'Submit Data';

            }

        });
